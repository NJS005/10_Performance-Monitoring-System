// Mock Data
const INITIAL_USER = {
  name: '',
  studentId: '',
  department: '',
  semester: '',
  email: 'student@institute.edu.in',
  profileComplete: false
};

function Student() {
  const [user, setUser] = useState(INITIAL_USER);
  const [activeTab, setActiveTab] = useState('overview');
  const [showProfileModal, setShowProfileModal] = useState(!user.profileComplete);
  const [showNotifications, setShowNotifications] = useState(false);
  const [mobileMenuOpen, setMobileMenuOpen] = useState(false);
  
  useEffect(() => {
    const profileComplete = localStorage.getItem('profileComplete');
    if (profileComplete === 'true') {
      const savedUser = JSON.parse(localStorage.getItem('user'));
      setUser(savedUser);
      setShowProfileModal(false);
    }
  }, []);

  return (
    <div className="min-h-screen bg-gradient-to-br from-[#2d3561] via-[#3d4573] to-[#4d5585]">
      {showProfileModal && (
        <ProfileCompletionModal 
          user={user} 
          setUser={setUser} 
          setShowProfileModal={setShowProfileModal} 
        />
      )}
      
      <Header 
        user={user} 
        showNotifications={showNotifications}
        setShowNotifications={setShowNotifications}
        mobileMenuOpen={mobileMenuOpen}
        setMobileMenuOpen={setMobileMenuOpen}
      />
      
      <div className="flex h-[calc(100vh-72px)]">
        <Sidebar 
          activeTab={activeTab} 
          setActiveTab={setActiveTab}
          mobileMenuOpen={mobileMenuOpen}
          setMobileMenuOpen={setMobileMenuOpen}
        />
        
        <main className="flex-1 overflow-y-auto bg-gray-50 p-8">
          {activeTab === 'overview' && <Overview user={user} />}
          {activeTab === 'grades' && <GradeManagement />}
          {activeTab === 'attendance' && <AttendanceTracker />}
          {activeTab === 'calculator' && <TargetCalculator />}
          {activeTab === 'chatbot' && <AcademicChatbot />}
        </main>
      </div>
      
      <ChatbotWidget />
    </div>
  );
}

// Profile Completion Modal
function ProfileCompletionModal({ user, setUser, setShowProfileModal }) {
  const [formData, setFormData] = useState({
    name: '',
    studentId: '',
    department: '',
    semester: ''
  });
  const [errors, setErrors] = useState({});

  const departments = [
    'Computer Science',
    'Information Technology',
    'Electronics & Communication',
    'Mechanical Engineering',
    'Civil Engineering',
    'Electrical Engineering'
  ];

  const validateForm = () => {
    const newErrors = {};
    
    if (!formData.name || formData.name.length < 3) {
      newErrors.name = 'Enter valid name (min 3 characters)';
    }
    
    if (!formData.studentId.match(/^\d{4}[A-Z]{2,4}\d{3,4}$/)) {
      newErrors.studentId = 'Invalid ID format (e.g., 2021CS001)';
    }
    
    if (!formData.department) {
      newErrors.department = 'Please select your department';
    }
    
    if (!formData.semester || formData.semester < 1 || formData.semester > 8) {
      newErrors.semester = 'Select semester between 1 and 8';
    }
    
    return newErrors;
  };

  const handleSubmit = (e) => {
    e.preventDefault();
    const newErrors = validateForm();
    
    if (Object.keys(newErrors).length === 0) {
      const updatedUser = { ...user, ...formData, profileComplete: true };
      setUser(updatedUser);
      localStorage.setItem('user', JSON.stringify(updatedUser));
      localStorage.setItem('profileComplete', 'true');
      setShowProfileModal(false);
    } else {
      setErrors(newErrors);
    }
  };

  return (
    <div className="fixed inset-0 bg-black/50 flex items-center justify-center z-50 p-6 animate-fadeIn">
      <div className="bg-white rounded-3xl max-w-2xl w-full max-h-[90vh] overflow-y-auto shadow-2xl animate-scaleIn">
        <div className="p-8 border-b border-gray-200">
          <h2 className="text-3xl font-bold text-gray-900 mb-2">Complete Your Profile</h2>
          <p className="text-gray-600">Required before accessing dashboard</p>
        </div>
        
        <form onSubmit={handleSubmit} className="p-8 space-y-6">
          <div>
            <label className="block text-sm font-semibold text-gray-700 mb-2">Institute Email (Read-Only)</label>
            <input 
              type="email" 
              value={user.email} 
              disabled 
              className="w-full px-4 py-3 bg-gray-100 border border-gray-300 rounded-xl text-gray-500 cursor-not-allowed"
            />
          </div>
          
          <div>
            <label className="block text-sm font-semibold text-gray-700 mb-2">Full Name *</label>
            <input 
              type="text" 
              placeholder="As per official records"
              value={formData.name}
              onChange={(e) => setFormData({...formData, name: e.target.value})}
              className={`w-full px-4 py-3 border rounded-xl transition-all ${
                errors.name ? 'border-red-500 focus:ring-red-200' : 'border-gray-300 focus:border-[#5b6ba8] focus:ring-4 focus:ring-[#5b6ba8]/20'
              } outline-none`}
            />
            {errors.name && <span className="text-red-600 text-sm mt-1 block">{errors.name}</span>}
          </div>
          
          <div>
            <label className="block text-sm font-semibold text-gray-700 mb-2">Student ID *</label>
            <input 
              type="text" 
              placeholder="Format: 2021CS001"
              value={formData.studentId}
              onChange={(e) => setFormData({...formData, studentId: e.target.value.toUpperCase()})}
              className={`w-full px-4 py-3 border rounded-xl transition-all ${
                errors.studentId ? 'border-red-500 focus:ring-red-200' : 'border-gray-300 focus:border-[#5b6ba8] focus:ring-4 focus:ring-[#5b6ba8]/20'
              } outline-none`}
            />
            {errors.studentId && <span className="text-red-600 text-sm mt-1 block">{errors.studentId}</span>}
          </div>
          
          <div>
            <label className="block text-sm font-semibold text-gray-700 mb-2">Department *</label>
            <select 
              value={formData.department}
              onChange={(e) => setFormData({...formData, department: e.target.value})}
              className={`w-full px-4 py-3 border rounded-xl transition-all ${
                errors.department ? 'border-red-500 focus:ring-red-200' : 'border-gray-300 focus:border-[#5b6ba8] focus:ring-4 focus:ring-[#5b6ba8]/20'
              } outline-none`}
            >
              <option value="">Select Department</option>
              {departments.map(dept => (
                <option key={dept} value={dept}>{dept}</option>
              ))}
            </select>
            {errors.department && <span className="text-red-600 text-sm mt-1 block">{errors.department}</span>}
          </div>
          
          <div>
            <label className="block text-sm font-semibold text-gray-700 mb-2">Current Semester *</label>
            <select 
              value={formData.semester}
              onChange={(e) => setFormData({...formData, semester: e.target.value})}
              className={`w-full px-4 py-3 border rounded-xl transition-all ${
                errors.semester ? 'border-red-500 focus:ring-red-200' : 'border-gray-300 focus:border-[#5b6ba8] focus:ring-4 focus:ring-[#5b6ba8]/20'
              } outline-none`}
            >
              <option value="">Select Semester</option>
              {[1,2,3,4,5,6,7,8].map(sem => (
                <option key={sem} value={sem}>Semester {sem}</option>
              ))}
            </select>
            {errors.semester && <span className="text-red-600 text-sm mt-1 block">{errors.semester}</span>}
          </div>
          
          <div className="pt-6 border-t border-gray-200">
            <button 
              type="submit" 
              className="w-full bg-[#5b6ba8] hover:bg-[#4d5a8f] text-white font-semibold py-4 rounded-xl transition-all transform hover:-translate-y-0.5 hover:shadow-lg"
            >
              Complete Profile
            </button>
          </div>
        </form>
      </div>
    </div>
  );
}

// Sidebar Component
function Sidebar({ activeTab, setActiveTab, mobileMenuOpen, setMobileMenuOpen }) {
  const menuItems = [
    { id: 'overview', label: 'Dashboard', icon: <TrendingUp size={20} /> },
    { id: 'grades', label: 'Grade Cards', icon: <Upload size={20} /> },
    { id: 'attendance', label: 'Attendance', icon: <Calendar size={20} /> },
    { id: 'calculator', label: 'CGPA Calculator', icon: <Target size={20} /> },
    { id: 'chatbot', label: 'Academic Guide', icon: <MessageCircle size={20} /> }
  ];

  const handleTabClick = (id) => {
    setActiveTab(id);
    setMobileMenuOpen(false);
  };

  return (
    <aside className={`w-64 bg-white border-r border-gray-200 p-6 transition-transform duration-300 lg:translate-x-0 ${
      mobileMenuOpen ? 'translate-x-0' : '-translate-x-full'
    } fixed lg:relative h-full z-30`}>
      <nav className="space-y-1">
        {menuItems.map(item => (
          <button
            key={item.id}
            className={`w-full flex items-center gap-4 px-4 py-3 rounded-xl font-medium text-sm transition-all ${
              activeTab === item.id 
                ? 'bg-[#5b6ba8]/10 text-[#5b6ba8] font-semibold' 
                : 'text-gray-700 hover:bg-gray-100'
            }`}
            onClick={() => handleTabClick(item.id)}
          >
            <span className={activeTab === item.id ? 'text-[#5b6ba8]' : 'text-gray-500'}>{item.icon}</span>
            <span>{item.label}</span>
          </button>
        ))}
      </nav>
    </aside>
  );
}

                    <button className="px-3 py-1 border border-gray-300 rounded-lg text-sm font-semibold hover:bg-gray-50">View</button>

// Target Calculator Component
function TargetCalculator() {
  const [formData, setFormData] = useState({
    currentCGPA: 8.25,
    creditsCompleted: 120,
    currentSemesterCredits: 24,
    targetCGPA: 8.50
  });
  const [result, setResult] = useState(null);

  const calculateRequiredSGPA = () => {
    const { currentCGPA, creditsCompleted, currentSemesterCredits, targetCGPA } = formData;
    const totalCreditsAfter = creditsCompleted + currentSemesterCredits;
    const requiredSGPA = ((targetCGPA * totalCreditsAfter) - (currentCGPA * creditsCompleted)) / currentSemesterCredits;
    
    let feasibility = 'achievable';
    if (requiredSGPA > 10) feasibility = 'impossible';
    else if (requiredSGPA > 9.5) feasibility = 'challenging';
    else if (requiredSGPA <= 0) feasibility = 'achieved';
    
    setResult({ requiredSGPA: requiredSGPA.toFixed(2), feasibility });
  };

  return (
    <div className="space-y-6">
      <div>
        <h1 className="text-4xl font-bold text-gray-900 mb-2">Target CGPA Calculator</h1>
        <p className="text-gray-600">Plan your academic goals and calculate required performance</p>
      </div>

      <div className="grid grid-cols-1 lg:grid-cols-2 gap-8">
        <div className="bg-white rounded-2xl shadow-sm overflow-hidden lg:sticky lg:top-24">
          <div className="p-6 border-b border-gray-200">
            <h2 className="text-xl font-semibold text-gray-900">Set Your Academic Goal</h2>
          </div>
          <div className="p-6 space-y-6">
            <div>
              <label className="block text-sm font-semibold text-gray-700 mb-2">Current CGPA *</label>
              <input 
                type="number" 
                step="0.01"
                value={formData.currentCGPA}
                onChange={(e) => setFormData({...formData, currentCGPA: parseFloat(e.target.value)})}
                className="w-full px-4 py-3 border border-gray-300 rounded-xl outline-none focus:border-[#5b6ba8] focus:ring-4 focus:ring-[#5b6ba8]/20"

              <input 
                type="number"
                step="0.01"
                value={formData.targetCGPA}
                onChange={(e) => setFormData({...formData, targetCGPA: parseFloat(e.target.value)})}
                className="w-full px-4 py-3 border border-gray-300 rounded-xl outline-none focus:border-[#5b6ba8] focus:ring-4 focus:ring-[#5b6ba8]/20"
              />
              <span className="text-xs text-gray-500 mt-1 block">Your desired CGPA goal</span>
            </div>

            <button 
              className="w-full flex items-center justify-center gap-2 px-6 py-4 bg-[#5b6ba8] text-white rounded-xl font-semibold hover:bg-[#4d5a8f] transition-all hover:shadow-lg"
              onClick={calculateRequiredSGPA}
            >
              <Target size={16} />
              Calculate Required SGPA
            </button>
          </div>
        </div>

        {result && (
          <div className={`rounded-2xl overflow-hidden shadow-lg border-2 animate-slideIn ${
            result.feasibility === 'achievable' ? 'border-green-500 bg-green-50' :
            result.feasibility === 'challenging' ? 'border-yellow-500 bg-yellow-50' :
            result.feasibility === 'impossible' ? 'border-red-500 bg-red-50' :
            'border-[#5b6ba8] bg-gradient-to-br from-green-50 to-blue-50'
          }`}>
            <div className={`p-6 border-b-2 ${
              result.feasibility === 'achievable' ? 'border-green-500' :
              result.feasibility === 'challenging' ? 'border-yellow-500' :
              result.feasibility === 'impossible' ? 'border-red-500' :
              'border-[#5b6ba8]'
            }`}>
              <h2 className="text-2xl font-bold">📊 Result: {result.feasibility.toUpperCase()}</h2>
            </div>
            <div className="p-8 space-y-6">
              <p className="text-lg">To achieve CGPA of <strong>{formData.targetCGPA}</strong>, you need:</p>
              
              <div className="flex items-center justify-between p-6 bg-white rounded-2xl shadow-sm">
                <span className="text-lg font-medium">Required SGPA:</span>
                <span className="text-5xl font-bold text-[#5b6ba8] font-mono">{result.requiredSGPA}</span>
              </div>

              {result.feasibility === 'achievable' && (
                <div className="flex gap-4 p-6 bg-white rounded-2xl">
                  <CheckCircle size={24} className="text-green-700 flex-shrink-0 mt-1" />
                  <div>
                    <strong className="block text-lg mb-2">This is achievable with focused effort!</strong>
                    <ul className="list-disc pl-5 space-y-1 text-gray-700">
                      <li>Aim for A/A+ grades in core subjects</li>
                      <li>Maintain &gt;85% attendance</li>
                      <li>Focus on high-weightage courses</li>
                    </ul>
                  </div>
                </div>
              )}

              {result.feasibility === 'challenging' && (
                <div className="flex gap-4 p-6 bg-white rounded-2xl">
                  <AlertCircle size={24} className="text-yellow-700 flex-shrink-0 mt-1" />
                  <div>
                    <strong className="block text-lg mb-2">This requires near-perfect performance</strong>
                    <ul className="list-disc pl-5 space-y-1 text-gray-700">
                      <li>Focus on high-weightage subjects</li>
                      <li>Seek additional guidance from faculty</li>
                      <li>Consider if this is realistic</li>
                    </ul>
                  </div>
                </div>
              )}

              {result.feasibility === 'impossible' && (
                <div className="flex gap-4 p-6 bg-white rounded-2xl">
                  <X size={24} className="text-red-700 flex-shrink-0 mt-1" />
                  <div>
                    <strong className="block text-lg mb-2">This target exceeds maximum SGPA (10.0)</strong>
                    <p className="mb-2">It's mathematically impossible this semester. Consider:</p>
                    <ul className="list-disc pl-5 space-y-1 text-gray-700">
                      <li>Lowering target to realistic value</li>
                      <li>Planning multi-semester strategy</li>
                    </ul>
                  </div>
                </div>
              )}

              {result.feasibility === 'achieved' && (
                <div className="flex gap-4 p-6 bg-white rounded-2xl">
                  <CheckCircle size={24} className="text-green-700 flex-shrink-0 mt-1" />
                  <div>
                    <strong className="block text-lg mb-2">Congratulations! Target already achieved!</strong>
                    <p>Your current CGPA ({formData.currentCGPA}) already exceeds this target. You can maintain or aim higher.</p>
                  </div>
                </div>
              )}

              <div className="flex gap-4">
                <button className="flex-1 px-6 py-3 border border-gray-300 bg-white rounded-xl font-semibold hover:bg-gray-50">
                  Save Target
                </button>
                <button className="flex-1 px-6 py-3 text-[#5b6ba8] hover:bg-[#5b6ba8]/10 rounded-xl font-semibold" onClick={() => setResult(null)}>
                  Try Another Target
                </button>
              </div>
            </div>
          </div>
        )}
      </div>
    </div>
  );
}

// Academic Chatbot Component
function AcademicChatbot() {
  const [messages, setMessages] = useState([
    {
      sender: 'bot',
      text: "Hi! I'm your academic advisor. I've analyzed your performance data. Here's what I found:",
      insights: [
        "📊 Your CGPA trend shows a 0.3 dip in Sem 4",
        "📉 Subjects needing attention: CS402, MA401",
        "💡 You excel in programming subjects"
      ],
      suggestions: [
        { label: 'Improve weak subjects', action: 'weakness' },
        { label: 'Suggest electives', action: 'electives' },
        { label: 'Analyze attendance', action: 'attendance' }
      ]
    }
  ]);
  const [inputText, setInputText] = useState('');

  const handleSend = () => {
    if (!inputText.trim()) return;
    
    setMessages([...messages, { sender: 'user', text: inputText }]);
    
    setTimeout(() => {
      setMessages(prev => [...prev, {
        sender: 'bot',
        text: "Based on your strong performance in CS501 (A+) and CS502 (A), I recommend these electives:",
        recommendations: [
          { name: 'Advanced Machine Learning', reason: 'Builds on CS503 where you scored A+', relevance: 'High' },
          { name: 'Cloud Computing', reason: 'Complements your programming skills', relevance: 'Very High' },
          { name: 'Blockchain Technology', reason: 'Emerging field matching your interests', relevance: 'Medium' }
        ]
      }]);
    }, 1000);
    
    setInputText('');
  };

  return (
    <div className="space-y-6">
      <div>
        <h1 className="text-4xl font-bold text-gray-900 mb-2">🤖 Academic Guidance Assistant</h1>
        <p className="text-gray-600">Get personalized insights based on your academic performance</p>
      </div>

      <div className="bg-white rounded-2xl shadow-lg overflow-hidden h-[600px] flex flex-col">
        <div className="flex-1 overflow-y-auto p-8 space-y-6">
          {messages.map((msg, idx) => (
            <div key={idx} className={`flex gap-4 animate-fadeInUp ${msg.sender === 'user' ? 'justify-end' : ''}`}>
              {msg.sender === 'bot' && (
                <div className="w-10 h-10 rounded-full bg-gradient-to-br from-[#5b6ba8] to-[#4d5a8f] flex items-center justify-center text-white flex-shrink-0 text-xl">
                  🤖
                </div>
              )}
              <div className={`max-w-[80%] ${msg.sender === 'user' ? 'max-w-[70%]' : ''}`}>
                <div className={`p-4 rounded-2xl ${
                  msg.sender === 'user' 
                    ? 'bg-[#5b6ba8] text-white rounded-br-none' 
                    : 'bg-gray-100 rounded-bl-none'
                }`}>
                  <p className="mb-0">{msg.text}</p>
                  
                  {msg.insights && (
                    <ul className="mt-4 space-y-2">
                      {msg.insights.map((insight, i) => (
                        <li key={i} className="p-2 bg-white rounded-lg text-gray-900 text-sm">{insight}</li>
                      ))}
                    </ul>
                  )}
                  
                  {msg.suggestions && (
                    <div className="mt-4 pt-4 border-t border-gray-200">
                      <p className="font-semibold mb-2">What would you like to explore?</p>
                      <div className="flex flex-wrap gap-2">
                        {msg.suggestions.map((sug, i) => (
                          <button key={i} className="px-4 py-2 bg-white border border-gray-300 rounded-xl text-sm hover:bg-[#5b6ba8]/10 hover:border-[#5b6ba8] hover:text-[#5b6ba8] transition-all">
                            {sug.label}
                          </button>
                        ))}
                      </div>
                    </div>
                  )}
                  
                  {msg.recommendations && (
                    <div className="mt-4 space-y-3">
                      {msg.recommendations.map((rec, i) => (
                        <div key={i} className="p-4 bg-white border border-gray-200 rounded-2xl">
                          <h4 className="font-semibold text-[#5b6ba8] mb-2">{i + 1}. {rec.name}</h4>
                          <p className="text-sm text-gray-700 mb-2"><strong>Why:</strong> {rec.reason}</p>
                          <span className="inline-block px-3 py-1 bg-green-100 text-green-700 rounded-lg text-xs font-semibold mb-3">
                            Career Relevance: {rec.relevance}
                          </span>
                          <div className="flex gap-2">
                            <button className="px-3 py-1 border border-gray-300 rounded-lg text-sm font-semibold hover:bg-gray-50">
                              View Syllabus
                            </button>
                            <button className="px-3 py-1 border border-gray-300 rounded-lg text-sm font-semibold hover:bg-gray-50">
                              Prerequisites
                            </button>
                          </div>
                        </div>
                      ))}
                    </div>
                  )}
                </div>
              </div>
            </div>
          ))}
        </div>

        <div className="flex gap-4 p-6 border-t border-gray-200 bg-gray-50">
          <input 
            type="text" 
            placeholder="Type your question..."
            value={inputText}
            onChange={(e) => setInputText(e.target.value)}
            onKeyPress={(e) => e.key === 'Enter' && handleSend()}
            className="flex-1 px-4 py-3 border border-gray-300 rounded-2xl outline-none focus:border-[#5b6ba8] focus:ring-4 focus:ring-[#5b6ba8]/20"
          />
          <button 
            className="px-6 py-3 bg-[#5b6ba8] text-white rounded-2xl font-semibold hover:bg-[#4d5a8f] transition-colors"
            onClick={handleSend}
          >
            Send
          </button>
        </div>
      </div>
    </div>
  );
}
